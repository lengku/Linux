pip 高级用法
	yum -y install python-pip
	install: 		安装软件
	download：		下载安装包
	uninstall：		卸载安装包
	freeze:			按照requirements格式输出按照包
	list：			列出当前系统中的安装包
	show			查看安装包
	check			检查依赖性
	search			查找
	completion		生成命令补全配置
	增加pip加速源
		pip install -i https://pypi.douban.com/simple/ flask
		更改配置文件
			cat >>~/.pip/pip.conf<<EOF
			[global]
			index-url = https://pypi.douban.com/simple/
			EOF

    安装方法
		下载到本地
		pip download flask
		pip install --no-index --find-links="/soft/flask" flask
		pip install --download='pwd' -r requirements.txt
		本地安装
		pip install --no-index -f file://'pwd' -r requirements.txt
