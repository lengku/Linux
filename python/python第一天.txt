									install python

1.安装相关的依赖
	yum -y install openssl-devel readline-devel unzip
2.下载源码
	wget https://www.python.org/ftp/python/3.6.2/Python-3.6.2.tgz
	tar xzf Python-3.6.2.tgz
	cd Python-3.6.2
	./configure --prefix=/usr/local/python36
	make
	make install



3.配置pip
  	more /etc/pip.conf
		[global]
		index-url = http://pypi.douban.com/simple
		trusted-host = pypi.douban.com
		[list]
		format=columns

4.install virtualenv
	/usr/local/python36/bin/python3.6 -m pip install --upgrade pip
	/usr/local/python36/bin/pip3 install virtualenv
5.start
	/usr/local/python36/bin/virtualenv ./python36env
	source  python36env/bin/activate
	pip install "django>1.11,<=1.12"
	pip install ipthon
	#install mysql
	pip install pymysql   #python3
